import * as React from "react";
import { Route, Link } from 'react-router-dom';
import { History } from 'history';
import * as _ from 'lodash';
import * as classnames from 'classnames';
 
import { Button, notification } from 'antd';
 
 
import { ISocketInitVideoCallRequest, ISocketMentoringData, IMentoring, ISocketRequestVideoCallResponse, ISocketAnswerVideoCallRequest, ISocketRequestVideoCallRequest } from '../../../../../../core/interfaces/barrel-interfaces';
import { ROLE_TYPE, SOCKET_REQUEST_TYPES } from '../../../../../../core/constants/barrel-constants';
 
import { CallWindowComponent } from './call-window.component';
import { MainWindowComponent } from './main-window.component';
import { CallModalComponent } from './call-modal.component';

import { PeerConnectionService } from './peer-connection.service';

import { SocketCommunicationService } from '../../../shared/services/barrel-services';
import { ROUTES_PATHS } from '../../../shared/constants/barrel-constants';
 
import { Store } from '../../../app.store';

interface IVideoCallState {
	clientId: string;
	callWindow: string;
	callModal: string;
	callModalType: string;
	callFrom: string;
	localSrc: string;
	peerSrc: string;
	isDuetOnline: boolean;
	actualMentoring: IMentoring;
}

 
export class VideoCallContainer extends React.Component<any, IVideoCallState> {
 
	private pc: PeerConnectionService;
	private startCallHandler: any;
	private endCallHandler: any;
	private rejectCallHandler: any;
	private socket: SocketIOClient.Socket;

	constructor(props: any) {
	  super(props);
	  this.resetState();
	  this.startCallHandler = this.startCall.bind(this);
	  this.endCallHandler = this.endCall.bind(this);
	  this.rejectCallHandler = this.rejectCall.bind(this);
	}

	componentWillUnmount(): void {
		console.log('component vai desmontar');
		this.setState({ isDuetOnline: false });
	}

	componentDidMount(): void {
		console.log('component vai montar');
		this.resetState();
		// Get mentoring
		// this.setState({ actualMentoring: Store.getStore().user.mentorings[0] });

		// Get IDs
		const userID: string = Store.getStore().user.primaryID;
		const duetID: string = this.state.actualMentoring.duet.primaryID;

		// Construct data that will be used through the requests
		const socketMentoringData: ISocketMentoringData = {
			mentoringPrimaryID: this.state.actualMentoring.primaryID,
			userRoleType: this.state.actualMentoring.mentoringRole,
			userID: userID,
			duetID: duetID,
			mentorID: this.state.actualMentoring.mentoringRole == ROLE_TYPE.MENTOR ? userID : duetID,
			mentoredID: this.state.actualMentoring.mentoringRole == ROLE_TYPE.MENTORED ? userID : duetID,
		};

		// Get socket reference
		this.socket = SocketCommunicationService.factorySocket();

		// Init request
		const initRequest: ISocketInitVideoCallRequest = {
			mentoringData: socketMentoringData
		};
 
	  this.socket
			.on('connect', () => console.log('conectei - client'))
			.on(SOCKET_REQUEST_TYPES.reconnect, () => 
				this.socket.emit(SOCKET_REQUEST_TYPES.init, initRequest))
	    .on(SOCKET_REQUEST_TYPES.init, () => this.setState({ clientId: Store.getStore().user.primaryID }))
	  	.on(SOCKET_REQUEST_TYPES.peerConnected, () => {
	  		console.log('peerConnected');
	  		this.setState({ isDuetOnline: true });
	  	})
	  	.on(SOCKET_REQUEST_TYPES.peerDisconnected, () => {
				console.log('peerDisconnected');
				notification.open({
					message: 'Par desconectado',
					description: 'Seu par se desconectou, aguarde até que ele se reconecte',
				});
	  		this.setState({ isDuetOnline: false, callModal: '', callModalType: '' });
	  	})
			.on(SOCKET_REQUEST_TYPES.request, (data: ISocketRequestVideoCallResponse) => {
				console.log('request', data);
				this.setState({ callModal: 'active', callModalType: 'request', callFrom: data.from });
			})
			.on(SOCKET_REQUEST_TYPES.answer, (data: ISocketAnswerVideoCallRequest) => {
				this.setState({ callModal: '', callModalType: '' });
				if (data.positiveAnswer) this.startCall(true);
				else this.endCall(true);
			})
	    .on(SOCKET_REQUEST_TYPES.call, (data: any) => {
	      if (data.sdp && this.pc) {
	        this.pc.setRemoteDescription(data.sdp);
	        if (data.sdp.type === 'offer') this.pc.createAnswer();
	      }
	      else if (this.pc) this.pc.addIceCandidate(data.candidate);
	    })
	    .on(SOCKET_REQUEST_TYPES.end, this.endCall.bind(this, false))
	    .emit(SOCKET_REQUEST_TYPES.init, initRequest);
	}

	waitForAnswer() {
		const req: ISocketRequestVideoCallRequest = {
			to: Store.getStore().user.mentorings[0].duet.primaryID
		};
		this.socket.emit(SOCKET_REQUEST_TYPES.request, req);
		this.setState({ callModal: 'active', callModalType: 'answer' });
		console.log('Esperando resposta');
	}

	acceptCall() {
		const req: ISocketAnswerVideoCallRequest = {
			positiveAnswer: true,
			to: this.state.callFrom
		};
	  this.socket.emit(SOCKET_REQUEST_TYPES.answer, req);
		this.setState({ callModal: '', callModalType: '' });
		this.startCall(false);
	}
 
	rejectCall() {
		const req: ISocketAnswerVideoCallRequest = {
			positiveAnswer: false,
			to: this.state.callFrom
		};
	  this.socket.emit(SOCKET_REQUEST_TYPES.answer, req);
	  this.socket.emit(SOCKET_REQUEST_TYPES.end, { to: this.state.callFrom });
		this.setState({ callModal: '', callModalType: '' });
		this.endCall(false);
	}

	startCall(isCaller: boolean) {
		console.log('startCall');
	  this.pc = new PeerConnectionService(Store.getStore().user.mentorings[0].duet.primaryID)
	    .on('localStream', (src: any) => {
	      const newState = { callWindow: 'active', localSrc: src, callModal: '', callModalType: '' };
	      if (!isCaller) newState.callModal = '';
	      this.setState(newState);
	    })
	    .on('peerStream', (src: any) => this.setState({ peerSrc: src }))
			.start(isCaller);
	}
 
	endCall(isStarter: any) {
	  if (this.pc && _.isFunction(this.pc.stop)) this.pc.stop(isStarter);
	  // this.pc = {};
	  this.setState({
	    callWindow: '',
	    localSrc: null,
	    peerSrc: null
	  });
	  const history: History = Store.getStore().history;
	  history.push(ROUTES_PATHS.mentoriaEnd);
	}
 
	private resetState() {
		this.state = {
	    clientId: '',
	    callWindow: '',
	    callModal: '',
	    callModalType: '',
	    callFrom: '',
	    localSrc: null,
	    peerSrc: null,
			isDuetOnline: false,
	    actualMentoring: Store.getStore().user.mentorings[0]
	  };
	}
 
	render() {
	  return (
	    <div className="full-page">
	      <MainWindowComponent
	        clientId={this.state.clientId}
	        startCall={ () => this.waitForAnswer() }
	        isDuetOnline={this.state.isDuetOnline}
	        friendID={Store.getStore().user.mentorings[0].duet.primaryID}
	      />
	        {/*username={this.props.username}*/}
	      <CallWindowComponent
	        status={this.state.callWindow}
	        localSrc={this.state.localSrc}
	        peerSrc={this.state.peerSrc}
	        mediaDevice={this.pc ? this.pc.mediaDevice : null}
	        endCall={this.endCallHandler}
	      />
	      <CallModalComponent
					type={this.state.callModalType}
	        status={this.state.callModal}
	        startCall={ () => this.acceptCall() }
	        rejectCall={ () => this.rejectCall() }
	        callFrom={this.state.callFrom}
	      />
	    </div >
	  );
	}
}
