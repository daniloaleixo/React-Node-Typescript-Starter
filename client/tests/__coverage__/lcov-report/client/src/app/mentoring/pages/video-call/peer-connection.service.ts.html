import { MediaDeviceService } from './media-device.service';
import { EmitterService } from './emitter.service';
import { SocketCommunicationService } from '../../../shared/services/socket-communication.service';
 
const PC_CONFIG = { iceServers: [{ urls: ['stun:stun.l.google.com:19302'] }] };
 
export class PeerConnectionService extends EmitterService {
 
  private pc: RTCPeerConnection;
  private friendID: string;
  public mediaDevice: MediaDeviceService;
  private socket: SocketIOClient.Socket = undefined;

  /**
   * Create a PeerConnection.
   * @param {String} friendID - ID of the friend you want to call.
   */
  constructor(friendID: string) {
    super();
 
    this.socket = SocketCommunicationService.factorySocket();
    this.pc = new RTCPeerConnection(PC_CONFIG);
    this.pc.onicecandidate = (event: any) => this.socket.emit('call', {
      to: this.friendID,
      candidate: event.candidate
    });
    this.pc.onaddstream = (event: any) => this.emit('peerStream', event.stream);
 
    this.mediaDevice = new MediaDeviceService();
    this.friendID = friendID;
  }
  /**
   * Starting the call
   * @param {Boolean} isCaller
   * @param {Object} config - configuration for the call {audio: boolean, video: boolean}
   */
  start(isCaller: any) {
    this.mediaDevice
      .on('stream', (stream: any) => {
        this.pc.addStream(stream);
        this.emit('localStream', stream);
        // if (isCaller) this.socket.emit('request', { to: this.friendID });
        if (!isCaller) this.createOffer();
      })
      .start();

    return this;
  }
  /**
   * Stop the call
   * @param {Boolean} isStarter
   */
  stop(isStarter: any) {
    if (isStarter) this.socket.emit('end', { to: this.friendID });
    this.mediaDevice.stop();
    this.pc.close();
    this.off();
    return this;
  }

  createOffer() {
    this.pc.createOffer()
      .then(this.getDescription.bind(this))
      .catch((err: any) => console.log(err));
    return this;
  }

  createAnswer() {
    this.pc.createAnswer()
      .then(this.getDescription.bind(this))
      .catch((err: any) => console.log(err));
    return this;
  }

  getDescription(desc: any) {
    this.pc.setLocalDescription(desc);
    this.socket.emit('call', { to: this.friendID, sdp: desc });
    return this;
  }
 
  /**
   * @param {Object} sdp - Session description
   */
  setRemoteDescription(sdp: any) {
    const rtcSdp = new RTCSessionDescription(sdp);
    this.pc.setRemoteDescription(rtcSdp);
    return this;
  }
  /**
   * @param {Object} candidate - ICE Candidate
   */
  addIceCandidate(candidate: any) {
    if (candidate) {
      const iceCandidate = new RTCIceCandidate(candidate);
      this.pc.addIceCandidate(iceCandidate);
    }
    return this;
  }
}
