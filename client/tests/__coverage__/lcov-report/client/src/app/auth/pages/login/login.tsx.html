import * as React from "react";
import { Route, Link, withRouter, RouteComponentProps } from 'react-router-dom';
import { GoogleLogin, GoogleLoginResponse } from 'react-google-login';
 
import { Loading } from '../../../shared/components/barrel-components';
import { ROUTES_PATHS } from '../../../shared/constants/barrel-constants';
import { AuthService } from '../../../shared/services/auth.service';
import { config } from '../../../shared/config/config';
 
import { IRegisterRequest, ILoginRequest, ILoginRegisterResponse, ILTFError, ILoginGoogleRequest, IGoogleAuthUser, IGoogleAuthResponse } from '../../../../../../core/interfaces/barrel-interfaces';
import { requestsInfo } from '../../../../../../core/constants/barrel-constants';
import { ERROR_MESSAGES } from '../../../../../../core/aggregators/error.aggregator';
 
// Assets
import './login.scss';
const logo = require('../../../../assets/img/logo-white.svg');

import { Modal, Button } from 'antd';
 
 
interface ILoginState {
  username: string;
  password: string;
  email: string;
  showLoading: boolean;
  isRegistering: boolean;
  errorMessage: string;
}

class LoginContainer extends React.Component<RouteComponentProps<string>, ILoginState> {
 
  constructor(props: RouteComponentProps<string>) {
    super(props);
    this.state = {
      username: '',
      password: '',
      email: '',
      showLoading: false,
      isRegistering: false,
      errorMessage: ''
    };
  }
 
  public login(e: any): void {
    e.preventDefault();
 
    this.setState({ showLoading: true });

    const req: ILoginRequest = {
        usernameOrEmail: this.state.email,
        password: this.state.password,
    };
 
    this.abstractLoginRegisterRequest(AuthService.login(req));
  }
 
  public googleLogin(user: IGoogleAuthResponse): void {
    this.setState({ showLoading: true });
 
    const req: ILoginGoogleRequest = {
      googleUser: {
        primaryID: null,
        username: user.profileObj.name,
        ...user.profileObj
      }
    };
    this.abstractLoginRegisterRequest(AuthService.loginWithGoogle(req));
  }

  /**
   * Register function
   * @param {*} e 
   * @memberof LoginContainer
   */
  public register(e: any): void {
    e.preventDefault();

    this.setState({ showLoading: true});
 
    const req: IRegisterRequest = {
      user: {
        username: this.state.username,
        password: this.state.password,
        email: this.state.email,
        primaryID: '',
        mentorings: [],
        blocksAnswers: []
      }
    };

    this.abstractLoginRegisterRequest(AuthService.register(req));
  }
 
  /**
   * Abstract the login and register request, because both update state the same way
   * @private
   * @param {Promise<void>} req 
   * @memberof LoginContainer
   */
  private abstractLoginRegisterRequest(req: Promise<void>): void {
    req
    .then(() => {
      this.setState({
        showLoading: false,
        errorMessage: ''
      });
      this.props.history.push(ROUTES_PATHS.root);
    })
    .catch((err: ILTFError) => {
      this.setState({
        showLoading: false,
        errorMessage: err.message
      });
    });
  }
 
  public responseGoogleFailed(err: any): void {
    this.setState({
      showLoading: false,
      errorMessage: ERROR_MESSAGES.AUTH_05_couldNotLoginWithGoogle
    });
  }
 
  render() {
    return (
      <div className="login-page align-vertical-center">
        <Loading showLoading={ this.state.showLoading } />
        <div className="login-page__container form">
 
          <img src={logo} className="login-page__logo"/>
 
          { // If we have an error message we show it
            this.state.errorMessage ?
             <span className="login-page__error-message">{this.state.errorMessage}</span> : null
          }
 
          <form className="login-page__form">
 
            {
              // Input username
              this.state.isRegistering ?
              (
                <input
                type="text"
                placeholder="flyer123"
                value={this.state.username}
                onChange={ (evt) => this.setState({ username: evt.target.value })}/>
              )
              : null
            }
 
            <input type="email"
            placeholder="flyer@learntofly.com"
            value={this.state.email}
            onChange={ (evt) => this.setState({ email: evt.target.value })}/>
 
            <input type="password"
            placeholder="password"
            value={this.state.password}
            onChange={ (evt) => this.setState({ password: evt.target.value })}/>
 
 
            {
              // Button for register or login
              this.state.isRegistering ?
              (<button className="LTFbutton--pink"
              onClick={(e) => this.register(e)}>
                registar
              </button>)
              :
              (<button className="LTFbutton--pink"
                onClick={(e) => this.login(e)}>
                  entrar
              </button>)
            }
 
           <GoogleLogin className="LTFbutton--red"
              clientId={config.googleAuthURL}
              buttonText="Entrar com o Google"
              onSuccess={ (user: any) => this.googleLogin(user) }
              onFailure={ (err) => this.responseGoogleFailed(err) }
            />
 
            {/* For now I'm removing the ability to register the user will only login
            <p onClick={(e) => this.setState({ isRegistering: !this.state.isRegistering } )}>
              { this.state.isRegistering ? 'Já Possui conta' : 'Ainda não possui conta' }
            </p>*/}
          </form>
        </div>
      </div>
    );
  }
 
}
 
 
// Export with high order function of withRouter
export const Login: React.ComponentClass<Pick<RouteComponentProps<string>, any>> = withRouter(LoginContainer);
